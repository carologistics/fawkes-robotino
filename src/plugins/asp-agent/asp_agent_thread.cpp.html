/** @class AspAgentThread "asp_agent_thread.h"
 * The thread to start and control the Asp agent.
 */

/** Constructor. */
AspAgentThread::AspAgentThread() : Thread("AspAgentThread", Thread::OPMODE_WAITFORWAKEUP),
		BlockedTimingAspect(BlockedTimingAspect::WAKEUP_HOOK_THINK), asp::RefboxComm(logger, config)
{
	return;
}

/** Destructor. */
AspAgentThread::~AspAgentThread()
{
	return;
}

void
AspAgentThread::init()
{
	logger->log_info(name(), "Initialize ASP Agent");
	activateBeacon();
	openPublic();
	return;
}

void
AspAgentThread::loop()
{
	return;
}

void
AspAgentThread::finalize()
{
	logger->log_info(name(), "Finalize ASP Agent");
	return;
}
